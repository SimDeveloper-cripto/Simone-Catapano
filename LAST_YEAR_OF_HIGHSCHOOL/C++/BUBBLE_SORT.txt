#include <iostream>
using namespace std;

int main(void){
	int a[10];
	int n = 0;
	int i,j;
	
	do{
		cin >> n;
		if (n <=0 || n > 10) cout << "Errore!";
	}while(n <=0 || n > 10);
	
	for(i = 0; i < n; i++){
		cout << "Inserisci: ";
		cin >> a[i];
	}
	
	for(i = 0; i < n; i++){
		cout << a[i] << " ";
	}
	
	cout << endl << "Vettore ordinato\n";
	
	for (i = 0; i < n-1; i++){
		for (j = i+1; j < n; j++){
			if(a[i] > a[j]){
				int tmp;
				tmp = a[i];
				a[i] = a[j];
				a[j] = tmp;
			}
		}
	}
	
	for (i = 0; i < n; i++) cout << a[i] << endl;
}
